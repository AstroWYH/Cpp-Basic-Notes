# 关于#include的一些理解

```c
/* ------------------------------------------------------------------ */
.h内主要存放对外的接口
/* ------------------------------------------------------------------ */
#include "xxx" // 当前路径搜索
#include <xxx> // 系统路径搜索，如/usr/include等
#include <xxx.h> // 带不带.h不重要，是因为C++之前的某个版本历史原因导致，确保是文件就行
/* ------------------------------------------------------------------ */
#ifndef __SOMEFILE_H__ // 缺点：不同头文件的这个宏一样，就会翻车
#define __SOMEFILE_H__
// ...
#endif

#pragma once // 缺点：编译器提供保证，单位是一个文件，但如果某个.h有多份拷贝，编译器不能保证不被重复包含
/* ------------------------------------------------------------------ */
// a.h
int a; // 既是声明又是定义，2次导入时，会报错
extern int a; // 仅是声明，2次导入时，不会报错，但需要在别处定义，让编译器能找得到
void foo(); // 函数声明，2次导入时，不会报错
class A { // 类声明(可以说类只有这个，没有定义)，2次导入时，会报错

};
// b.h
#include "a.h"
// main.cpp
#include "a.h"
#include "b.h"
// 如果a.h内无宏保护，则会导入2次a.h的内容，那么什么时候会报错呢？
// 声明可以有多份，定义只能有1份
/* ------------------------------------------------------------------ */
```
